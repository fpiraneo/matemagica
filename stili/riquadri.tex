% RIQUADRO DEFINIZIONE
% Sfondo azzurro chiaro, bordo sinistro blu
% -------------------------------------------------------------
\newtcolorbox{definizione}[2][]{
    enhanced,
    breakable,
    colback     = coloreSecondario,
    colframe    = coloreAccento,
    leftrule    = 4pt,
    rightrule   = 0pt,
    toprule     = 0pt,
    bottomrule  = 0pt,
    arc         = 0pt,
    outer arc   = 0pt,
    fonttitle   = \bfseries\color{colorePrimarioScuro},
    title       = {\faBookOpen\quad Definizione: #2},
    attach boxed title to top left = {yshift=-2mm, xshift=4mm},
    boxed title style = {
        colback  = coloreAccento,
        colframe = coloreAccento,
        arc      = 2pt
    },
    top         = 8pt,
    bottom      = 8pt,
    left        = 8pt,
    right       = 8pt,
    #1
}
